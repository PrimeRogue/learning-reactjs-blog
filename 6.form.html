Thay vì phải setState() nhiều thành phần trong form như ở trên, ta có thể gôm các thành phần chung 1 object và sử dụng thuộc tính name 

            
function App() {
  // 1. Đầu tiên, khởi tạo values là 1 object. Nhớ đặt tên "phần tử trong object" giống với value của "thuộc tính name"
  const [values, setValues] = useState({
    fullname: "",
    email: "",
  });

  // 2. Sử dụng thuộc tính name
  const handleSetValue = (e) => {
    setValues({
      // 3. setState() sử dụng REPLACING nên phải clone
      ...values,
      [e.target.name]: e.target.value,
    });
  };
  return (
    <main>
      <form action="">
        <input
          type="text"
          name="fullname"
          placeholder="Enter fullname"
          onChange={handleSetValue}
        />
        <input
          type="email"
          name="email"
          placeholder="Enter email"
          onChange={handleSetValue}
        />
      </form>
    </main>
  );
}

2. Trường hợp thẻ input có type="checkbox"

            
function App() {
  const [values, setValues] = useState({
    fullname: "",
    email: "",
    single: false,
  });
  const handleSetValue = (e) => {
    // 1. Khai báo biến lấy type của thẻ input
    const typeInput = e.target.type;
    setValues({
      ...values,
      // 2. Kiểm tra type ucar input có phải checkbox hay không
      [e.target.name]:
        typeInput === "checkbox" ? e.target.checked : e.target.value,
    });
  };
  return (
    <main>
      <form action="">
        <input
          type="text"
          name="fullname"
          placeholder="Enter fullname"
          onChange={handleSetValue}
        />
        <input
          type="email"
          name="email"
          placeholder="Enter email"
          onChange={handleSetValue}
        />
        <input type="checkbox" name="single" onChange={handleSetValue} />
      </form>
    </main>
  );
}
